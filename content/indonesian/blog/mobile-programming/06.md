## Komponen TextField untuk Input dan Output
Aplikasi ini menggunakan komponen TextField untuk menerima input angka dari pengguna. Kita menggunakan KeyboardOptions(keyboardType = KeyboardType.Decimal) agar keyboard yang muncul di layar ponsel khusus untuk angka dan desimal.

Untuk menampilkan hasil perhitungan, kita menggunakan TextField tambahan namun dengan argumen readOnly = true agar pengguna tidak bisa mengetik di kolom hasil.
```java
TextField(
    value = num1,
    onValueChange = { num1 = it },
    label = { Text("Enter Number 1") },
    keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
    singleLine = true,
    modifier = Modifier.fillMaxWidth()
)

// TextField untuk Hasil
TextField(
    value = result,
    onValueChange = {},
    label = { Text("Result") },
    readOnly = true,
    modifier = Modifier.fillMaxWidth()
)
```

<br>

## Tombol Operasi
Untuk membuat tombol operasi penjumlahan, pengurangan, perkalian, dan pembagian tersusun sejajar secara horizontal, kita menggunakan komponen Row.
```java
val ops = listOf("+", "-", "*", "/")
ops.forEach { op ->
    Button(
        onClick = { result = calculateResult(num1, num2, op) },
        modifier = Modifier
            .weight(1f)
            .padding(2.dp),
        colors = ButtonDefaults.buttonColors(containerColor = Color(0xFFFF7043))
    ) {
        Text(op, fontSize = 18.sp)
    }
}
```

## Fungsi Logika Perhitungan
Logika utama kalkulator dipisahkan ke dalam fungsi calculateResult. Fungsi ini bertugas mengubah input String dari TextField menjadi tipe data Double agar bisa dihitung secara matematis. Fungsi ini juga menangani validasi pembagian dengan angka nol (0) dan membersihkan format desimal jika hasilnya adalah bilangan bulat.
```java
fun calculateResult(n1: String, n2: String, operation: String): String {
    val d1 = n1.toDoubleOrNull() ?: 0.0
    val d2 = n2.toDoubleOrNull() ?: 0.0

    val res = when (operation) {
        "+" -> d1 + d2
        "-" -> d1 - d2
        "*" -> d1 * d2
        "/" -> if (d2 != 0.0) d1 / d2 else return "Error (Div by 0)"
        else -> 0.0
    }

    // Clean up the output: if it's 7.0, show 7. If it's 7.5, show 7.5
    return if (res % 1 == 0.0) res.toInt().toString() else res.toString()
}
```

## Source Code
```java
package com.example.mansescalculator

import android.os.Bundle
import androidx.activity.ComponentActivity
import androidx.activity.compose.setContent
import androidx.activity.enableEdgeToEdge
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.text.KeyboardOptions
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Modifier
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.text.font.FontWeight
import androidx.compose.ui.text.input.KeyboardType
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp
import com.example.mansescalculator.ui.theme.MansesCalculatorTheme

class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContent {
            MansesCalculatorTheme {
                Scaffold(modifier = Modifier.fillMaxSize()) { innerPadding ->
                    MainCalculator(
                        modifier = Modifier
                            .padding(innerPadding)
                    )
                }
            }
        }
    }
}

@Composable
fun MainCalculator(modifier: Modifier = Modifier) {
    var num1 by remember { mutableStateOf("") }
    var num2 by remember { mutableStateOf("") }
    var result by remember { mutableStateOf("") }

    Column(
        modifier = modifier
            .fillMaxSize()
            .padding(16.dp)
    ) {
        Text(
            text = "Manses Calculator",
            fontSize = 24.sp,
            fontWeight = FontWeight.Bold,
            modifier = Modifier.padding(bottom = 16.dp)
        )

        TextField(
            value = num1,
            onValueChange = { num1 = it },
            label = { Text("Enter Number 1") },
            keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
            singleLine = true,
            modifier = Modifier.fillMaxWidth()
        )

        Spacer(modifier = Modifier.height(12.dp))

        TextField(
            value = num2,
            onValueChange = { num2 = it },
            label = { Text("Enter Number 2") },
            keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
            singleLine = true,
            modifier = Modifier.fillMaxWidth()
        )

        Spacer(modifier = Modifier.height(12.dp))

        TextField(
            value = result,
            onValueChange = {},
            label = { Text("Result") },
            readOnly = true,
            modifier = Modifier.fillMaxWidth(),
            colors = TextFieldDefaults.colors(
                focusedContainerColor = Color(0xFFF1F1F1),
                unfocusedContainerColor = Color(0xFFF1F1F1)
            )
        )

        Spacer(modifier = Modifier.height(20.dp))

        // Button Row
        Row(
            modifier = Modifier.fillMaxWidth(),
            horizontalArrangement = Arrangement.SpaceBetween
        ) {
            val ops = listOf("+", "-", "*", "/")
            ops.forEach { op ->
                Button(
                    onClick = { result = calculateResult(num1, num2, op) },
                    modifier = Modifier
                        .weight(1f)
                        .padding(2.dp),
                    colors = ButtonDefaults.buttonColors(containerColor = Color(0xFFFF7043))
                ) {
                    Text(op, fontSize = 18.sp)
                }
            }

            Button(
                onClick = {
                    num1 = ""
                    num2 = ""
                    result = ""
                },
                modifier = Modifier
                    .weight(1.5f)
                    .padding(2.dp),
                colors = ButtonDefaults.buttonColors(containerColor = Color(0xFFFF7043))
            ) {
                Text("Clear")
            }
        }
    }
}

fun calculateResult(n1: String, n2: String, operation: String): String {
    val d1 = n1.toDoubleOrNull() ?: 0.0
    val d2 = n2.toDoubleOrNull() ?: 0.0

    val res = when (operation) {
        "+" -> d1 + d2
        "-" -> d1 - d2
        "*" -> d1 * d2
        "/" -> if (d2 != 0.0) d1 / d2 else return "Error (Div by 0)"
        else -> 0.0
    }

    // Clean up the output: if it's 7.0, show 7. If it's 7.5, show 7.5
    return if (res % 1 == 0.0) res.toInt().toString() else res.toString()
}
```
